        # case 1
        lf22 = fit_2n2.Exp2Fit22(np.array([[60.,80.],[30.,12.]]),direction=[1,0])
        (x1,y1) = (40,lf22.f(40))
        (x2,y2) = (lf22.g(y1),y1)

        (x1,y1) = (30,lf22.f(30))
        (x2,y2) = (lf22.g(y1),y1) 
        assert(x1 == x2 and y1 == y2)

        (x1,y1) = (60,lf22.f(60))
        (x2,y2) = (lf22.g(y1),y1) 
        assert(x1 == x2 and y1 == y2)
